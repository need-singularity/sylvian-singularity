\begin{abstract}
We prove that $\sigma(n)\varphi(n) = n\tau(n)$ if and only if $n \in \{1,6\}$,
providing a new characterization of the smallest perfect number using only the
four basic arithmetic functions $\sigma$, $\varphi$, $\tau$, and the identity.
The dual equation $\sigma(n)\tau(n) = n\varphi(n)$ characterizes $n=28$.
We establish a spectral gap: the ratio $R(n) = \sigma(n)\varphi(n)/(n\tau(n))$
lies in $\{3/4\} \cup \{1\} \cup [7/6, \infty)$, with both intervals $(3/4,1)$
and $(1,7/6)$ empty.
\end{abstract}

\section{Introduction}

Let $\sigma(n)$, $\tau(n)$, and $\varphi(n)$ denote the sum of divisors,
number of divisors, and Euler's totient function, respectively. Define
\[
  R(n) = \frac{\sigma(n)\varphi(n)}{n\,\tau(n)}.
\]

\begin{theorem}\label{thm:main}
$R(n) = 1$ if and only if $n \in \{1, 6\}$.
\end{theorem}

\begin{theorem}\label{thm:gap}
For all $n \geq 2$,
\[
  R(n) \in \left\{\tfrac{3}{4}\right\} \cup \{1\} \cup \left[\tfrac{7}{6}, \infty\right).
\]
In particular, both $(3/4, 1)$ and $(1, 7/6)$ are empty.
\end{theorem}

\begin{theorem}\label{thm:dual}
$\sigma(n)\tau(n) = n\varphi(n)$ if and only if $n \in \{1, 28\}$.
\end{theorem}

\section{Proof of Theorem~\ref{thm:main}}

Since $\sigma$, $\varphi$, $\tau$ are multiplicative and $n \mapsto n$ is
completely multiplicative, $R$ is multiplicative on squarefree integers and
factors as
\[
  R(n) = \prod_{p^a \| n} f(p,a), \quad
  f(p,a) = \frac{p^{a+1}-1}{p(a+1)}.
\]

\textbf{Step 1.} $f(2,1) = 3/4$ is the unique value below 1.

For $a=1$: $f(p,1) = (p^2-1)/(2p) \geq 4/3$ when $p \geq 3$.
For $a \geq 2$: $f(p,a) \geq f(2,2) = 7/6 > 1$.

\textbf{Step 2.} $R(n)=1$ requires $2 \| n$.

If $2 \nmid n$, all factors exceed 1. If $4 \mid n$, the 2-part contributes
$f(2,a) \geq 7/6 > 1$.

\textbf{Step 3.} The remaining product equals $4/3$.

We need $f(q,1) = 4/3$, giving $(q^2-1)/(2q) = 4/3$, so $3q^2 - 8q - 3 = 0$,
with unique prime solution $q = 3$.

\textbf{Step 4.} No further factors allowed.

Any additional $f(r,b) \geq 4/3$ pushes the product above $4/3$.
Hence $n = 2 \times 3 = 6$.  \qed

\section{Proof of Theorem~\ref{thm:gap}}

We show $R(n) \geq 7/6$ for $n \geq 3$, $n \neq 6$.

\textbf{Case A: $n$ odd.} All $f(p,a) \geq f(3,1) = 4/3 > 7/6$.

\textbf{Case B: $2 \| n$, $n \neq 6$.}
$R = \frac{3}{4} \cdot \prod f(q_j, b_j)$ with at least one odd factor $\neq f(3,1)$.
If the odd part is a single prime $q \geq 5$: $R \geq \frac{3}{4} \cdot \frac{12}{5} = \frac{9}{5} > \frac{7}{6}$.
If two or more odd primes: $R \geq \frac{3}{4} \cdot \left(\frac{4}{3}\right)^2 = \frac{4}{3} > \frac{7}{6}$.

\textbf{Case C: $4 \| n$.}
$f(2,2) = 7/6$. If $n=4$: $R = 7/6$ (equality). If $n = 4m$ with $m > 1$:
$R \geq \frac{7}{6} \cdot \frac{4}{3} > \frac{7}{6}$.

\textbf{Case D: $8 \mid n$.}
$f(2,a) \geq f(2,3) = 15/8 > 7/6$. \qed

\section{Proof of Theorem~\ref{thm:dual}}

Define $S(n) = \sigma(n)\tau(n)/(n\varphi(n))$. Then $R(n)S(n) = (\sigma(n)/n)^2$.
For perfect numbers, $\sigma/n = 2$, so $RS = 4$.
$S(n) = 1$ iff $R(n) = 4$ iff $\sigma(n)/n = 2$ and $\tau/\varphi = 1/2$.
For $n = 2^{p-1}(2^p-1)$: $\varphi/\tau = 2^{p-2}(2^{p-1}-1)/p$.
Setting this equal to $2$: $2^{p-2}(2^{p-1}-1) = 2p$, which gives $p = 3$,
hence $n = 28$.  \qed

\section{Further results}

\begin{proposition}
$R(n)S(n) = (\sigma(n)/n)^2$ for all $n$. In particular, $R = S$ iff $\varphi(n) = \tau(n)$.
\end{proposition}

\begin{proposition}
There are exactly $24$ distinct values of $R(n) < 5$, enumerable from local factors.
\end{proposition}

\begin{corollary}
For odd $n$ with $\omega(n)$ distinct prime factors,
$\varphi(n)/\tau(n) \geq (4/3)^{\omega(n)}/2$.
\end{corollary}

\begin{corollary}
$R(n!) \in \mathbb{Z}$ for $n \geq 2$ iff $n \in \{3, 5\}$.
Then $R(3!) = R(6) = 1$ and $R(5!) = R(120) = 6$.
\end{corollary}

\begin{corollary}
The only perfect number in the image of $R$ is $6$: $R(120) = 6$.
No $n$ satisfies $R(n) = 28$ or $R(n) = 496$.
\end{corollary}
