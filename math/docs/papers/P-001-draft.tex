\begin{abstract}
We prove that $\sigma(n)\varphi(n) = n\tau(n)$ if and only if $n \in \{1,6\}$,
providing a new characterization of the smallest perfect number using the
four basic arithmetic functions. The proof reduces to showing that
$f(p,a) = (p^{a+1}-1)/(p(a+1))$ satisfies $f(2,1)=3/4 < 1$ uniquely,
with exact cancellation $f(2,1)f(3,1) = (3/4)(4/3) = 1$ from consecutive primes.
The dual equation $\sigma(n)\tau(n) = n\varphi(n)$ characterizes $n=28$.
Defining $R(n) = \sigma(n)\varphi(n)/(n\tau(n))$, we establish:
(i) a spectral gap $R(n) \in \{3/4\} \cup \{1\} \cup [7/6, \infty)$;
(ii) $R(n) < n$ for all $n \geq 2$, so integer $R$-chains converge to 1;
(iii) $R(n)S(n) = (\sigma(n)/n)^2$ where $S$ is the dual ratio;
(iv) exactly 24 distinct values of $R(n) < 5$.
Three ``pointwise equals convolution'' identities yield finite solution sets
$\{1,6\}$, $\{1,3,14,42\}$, and $\{1,3,10,30\}$, with cancellation primes
$\{3,5,7\}$---the three smallest odd primes.
\end{abstract}

\section{Introduction}

Let $\sigma(n)$, $\tau(n)$, and $\varphi(n)$ denote the sum of divisors,
number of divisors, and Euler's totient function, respectively. Define
\[
  R(n) = \frac{\sigma(n)\varphi(n)}{n\,\tau(n)}.
\]

\begin{theorem}\label{thm:main}
$R(n) = 1$ if and only if $n \in \{1, 6\}$.
\end{theorem}

\begin{theorem}\label{thm:gap}
For all $n \geq 2$,
\[
  R(n) \in \left\{\tfrac{3}{4}\right\} \cup \{1\} \cup \left[\tfrac{7}{6}, \infty\right).
\]
In particular, both $(3/4, 1)$ and $(1, 7/6)$ are empty.
\end{theorem}

\begin{theorem}\label{thm:dual}
$\sigma(n)\tau(n) = n\varphi(n)$ if and only if $n \in \{1, 28\}$.
\end{theorem}

\section{Proof of Theorem~\ref{thm:main}}

Since $\sigma$, $\varphi$, $\tau$ are multiplicative and $n \mapsto n$ is
completely multiplicative, $R$ is multiplicative on squarefree integers and
factors as
\[
  R(n) = \prod_{p^a \| n} f(p,a), \quad
  f(p,a) = \frac{p^{a+1}-1}{p(a+1)}.
\]

\textbf{Step 1.} $f(2,1) = 3/4$ is the unique value below 1.

For $a=1$: $f(p,1) = (p^2-1)/(2p) \geq 4/3$ when $p \geq 3$.
For $a \geq 2$: $f(p,a) \geq f(2,2) = 7/6 > 1$.

\textbf{Step 2.} $R(n)=1$ requires $2 \| n$.

If $2 \nmid n$, all factors exceed 1. If $4 \mid n$, the 2-part contributes
$f(2,a) \geq 7/6 > 1$.

\textbf{Step 3.} The remaining product equals $4/3$.

We need $f(q,1) = 4/3$, giving $(q^2-1)/(2q) = 4/3$, so $3q^2 - 8q - 3 = 0$,
with unique prime solution $q = 3$.

\textbf{Step 4.} No further factors allowed.

Any additional $f(r,b) \geq 4/3$ pushes the product above $4/3$.
Hence $n = 2 \times 3 = 6$.  \qed

\section{Proof of Theorem~\ref{thm:gap}}

We show $R(n) \geq 7/6$ for $n \geq 3$, $n \neq 6$.

\textbf{Case A: $n$ odd.} All $f(p,a) \geq f(3,1) = 4/3 > 7/6$.

\textbf{Case B: $2 \| n$, $n \neq 6$.}
$R = \frac{3}{4} \cdot \prod f(q_j, b_j)$ with at least one odd factor $\neq f(3,1)$.
If the odd part is a single prime $q \geq 5$: $R \geq \frac{3}{4} \cdot \frac{12}{5} = \frac{9}{5} > \frac{7}{6}$.
If two or more odd primes: $R \geq \frac{3}{4} \cdot \left(\frac{4}{3}\right)^2 = \frac{4}{3} > \frac{7}{6}$.

\textbf{Case C: $4 \| n$.}
$f(2,2) = 7/6$. If $n=4$: $R = 7/6$ (equality). If $n = 4m$ with $m > 1$:
$R \geq \frac{7}{6} \cdot \frac{4}{3} > \frac{7}{6}$.

\textbf{Case D: $8 \mid n$.}
$f(2,a) \geq f(2,3) = 15/8 > 7/6$. \qed

\section{Proof of Theorem~\ref{thm:dual}}

Define $S(n) = \sigma(n)\tau(n)/(n\varphi(n))$. Then $R(n)S(n) = (\sigma(n)/n)^2$.
For perfect numbers, $\sigma/n = 2$, so $RS = 4$.
$S(n) = 1$ iff $R(n) = 4$ iff $\sigma(n)/n = 2$ and $\tau/\varphi = 1/2$.
For $n = 2^{p-1}(2^p-1)$: $\varphi/\tau = 2^{p-2}(2^{p-1}-1)/p$.
Setting this equal to $2$: $2^{p-2}(2^{p-1}-1) = 2p$, which gives $p = 3$,
hence $n = 28$.  \qed

\section{Further results}

\begin{proposition}
$R(n)S(n) = (\sigma(n)/n)^2$ for all $n$. In particular, $R = S$ iff $\varphi(n) = \tau(n)$.
\end{proposition}

\begin{proposition}
There are exactly $24$ distinct values of $R(n) < 5$, enumerable from local factors.
\end{proposition}

\begin{corollary}
For odd $n$ with $\omega(n)$ distinct prime factors,
$\varphi(n)/\tau(n) \geq (4/3)^{\omega(n)}/2$.
\end{corollary}

\begin{corollary}
$R(n!) \in \mathbb{Z}$ for $n \geq 2$ iff $n \in \{3, 5\}$.
Then $R(3!) = R(6) = 1$ and $R(5!) = R(120) = 6$.
\end{corollary}

\begin{corollary}
The only perfect number in the image of $R$ is $6$: $R(120) = 6$.
No $n$ satisfies $R(n) = 28$ or $R(n) = 496$.
\end{corollary}

\section{Three finite solution sets}

Applying the same ``pointwise = convolution value'' analysis to
other pairs of arithmetic functions:

\begin{center}
\begin{tabular}{lll}
\hline
Identity & Solutions where pointwise${}={}$convolution & Cancellation prime\\
\hline
$\sigma\varphi = n\tau$ & $\{1, 6\}$ & $q=3$ \\
$\tau\varphi = \sigma$  & $\{1, 3, 14, 42\}$ & $q=7$ (Mersenne) \\
$\varphi^2 = (\varphi*\varphi)$ & $\{1, 3, 10, 30\}$ & $q=5$ (Fermat) \\
\hline
\end{tabular}
\end{center}

The cancellation primes $\{3,5,7\}$ are the three smallest odd primes.
Three additional finite sets arise from simpler identities:
$n\varphi = (\mathrm{id}*\varphi) \Rightarrow \{1,4\}$,
$n\tau = (\mathrm{id}*\tau) \Rightarrow \{1,2\}$, and
$\varphi\lambda = (\varphi*\lambda) \Rightarrow \{1,4\}$.
All six sets contain $n=1$; all involve the prime $2$.
In each case, $f(2,1) < 1$ is the unique sub-unity local factor,
and the cancellation prime $q$ is determined by $f(2,1) \cdot f(q,1) = 1$.

\begin{theorem}\label{thm:decrease}
$R(n) < n$ for all $n \geq 2$. In particular, every integer $R$-chain
$n_0, n_1 = R(n_0), n_2 = R(n_1), \ldots$ converges to $1$.
\end{theorem}

\begin{remark}[Cantor-like spectrum]
The image of $R$ restricted to $[0,10]$ consists of only $63$
distinct rational values (for $n \leq 50000$),
with gaps covering $99.1\%$ of the interval.
The density increases: approximately $5$ values per unit near $R=5$,
growing to $9.2$ per unit near $R=100$.
The $6$-smooth sublattice $\{2^a 3^b\}$ gives a transparent
visualization: $R=1$ appears uniquely at $(a,b)=(1,1)$, i.e., $n=6$.
\end{remark}

\begin{proof}
For each prime power $p^a \| n$, we have
$f(p,a) = \frac{p^{a+1}-1}{p(a+1)} < p^a$,
since this is equivalent to $1 - p^{-(a+1)} < a+1$,
where the left side is less than $1$ and the right side is at least $2$.
Therefore $R(n) = \prod f(p_i, a_i) < \prod p_i^{a_i} = n$.
\end{proof}

\begin{theorem}\label{thm:S}
$S(n) > n$ if and only if $n = 2$ (where $S(2)=3$).
For $n \geq 3$, $S(n) < n$.
\end{theorem}

\begin{proof}
The local factor $g(p,a) = \sigma(p^a)\tau(p^a)/(p^a\varphi(p^a))$
satisfies $g(p,a) < p^a$ for all $(p,a) \neq (2,1)$,
since $g(p,1) = 2(p+1)/(p(p-1)) < p$ iff $2(p+1) < p^2(p-1)$,
which holds for $p \geq 3$.
The unique exception $g(2,1) = 3 > 2$ gives $S(2) = 3 > 2$.
For $n = 2m$ with $m \geq 2$: if $4 \mid n$ then $g(2,a) < 2^a$ for $a \geq 2$;
if $2 \| n$ then $S = 3 \cdot \prod g(q_j,b_j) < 3 \cdot n/2$,
and $3n/2 < 2n$ is trivially less than... more precisely,
$S(2p) = 3 \cdot g(p,1) < 2p$ since $6(p+1) < 2p^2(p-1)$ for $p \geq 3$.
\end{proof}
