\title{A characterization of 6 via arithmetic functions}
\author{}

\begin{abstract}
We characterize even perfect numbers via arithmetic functions in two ways:
\emph{multiplicatively}, $\sigma(n)\varphi(n) = n\tau(n)$ iff $n \in \{1,6\}$;
and \emph{additively}, $M_p(\sigma(n)+\varphi(n)) = (5 \cdot 2^{p-1}-3)n$
iff $n = 2^{p-1}(2^p-1)$ for each Mersenne prime $M_p = 2^p - 1$.
The multiplicative proof uses $f(2,1) f(3,1) = (3/4)(4/3) = 1$;
the additive proof uses $\sigma(P)/P = 2$ and monotonicity of
$c_1 A(k) + c_2 F(k)$ in prime factors of $k$.
We also prove: $\sigma\omega = n\tau$ iff $n=6$ (same as $\sigma\varphi = n\tau$
because $\varphi = \omega$ iff $n \in \{2,6\}$);
the dual $\sigma\tau = n\varphi$ iff $n = 28$;
a spectral gap $R(n) \in \{3/4\} \cup \{1\} \cup [7/6,\infty)$;
$R(n) < n$ for all $n \geq 2$;
an Euler factor closed form $E_p(2) = p\ln((p+1)/p) + 1/p$;
and the additive identity $\sigma + \varphi + \tau = 3n$
has infinitely many solutions $n = 2^a \cdot 3 \cdot (7 \cdot 2^{a-2} + a)$
whenever $7 \cdot 2^{a-2} + a$ is prime.
\end{abstract}

\section{Introduction}

Let $\sigma(n)$, $\tau(n)$, and $\varphi(n)$ denote the sum of divisors,
number of divisors, and Euler's totient function, respectively. Define
\[
  R(n) = \frac{\sigma(n)\varphi(n)}{n\,\tau(n)}.
\]

\begin{theorem}\label{thm:main}
$R(n) = 1$ if and only if $n \in \{1, 6\}$.
\end{theorem}

\begin{theorem}\label{thm:gap}
For all $n \geq 2$,
\[
  R(n) \in \left\{\tfrac{3}{4}\right\} \cup \{1\} \cup \left[\tfrac{7}{6}, \infty\right).
\]
In particular, both $(3/4, 1)$ and $(1, 7/6)$ are empty.
\end{theorem}

\begin{theorem}\label{thm:dual}
$\sigma(n)\tau(n) = n\varphi(n)$ if and only if $n \in \{1, 28\}$.
\end{theorem}

\section{Proof of Theorem~\ref{thm:main}}

Since $\sigma$, $\varphi$, $\tau$ are multiplicative and $n \mapsto n$ is
completely multiplicative, $R$ is multiplicative on squarefree integers and
factors as
\[
  R(n) = \prod_{p^a \| n} f(p,a), \quad
  f(p,a) = \frac{p^{a+1}-1}{p(a+1)}.
\]

\textbf{Step 1.} $f(2,1) = 3/4$ is the unique value below 1.

For $a=1$: $f(p,1) = (p^2-1)/(2p) \geq 4/3$ when $p \geq 3$.
For $a \geq 2$: $f(p,a) \geq f(2,2) = 7/6 > 1$.

\textbf{Step 2.} $R(n)=1$ requires $2 \| n$.

If $2 \nmid n$, all factors exceed 1. If $4 \mid n$, the 2-part contributes
$f(2,a) \geq 7/6 > 1$.

\textbf{Step 3.} The remaining product equals $4/3$.

We need $f(q,1) = 4/3$, giving $(q^2-1)/(2q) = 4/3$, so $3q^2 - 8q - 3 = 0$,
with unique prime solution $q = 3$.

\textbf{Step 4.} No further factors allowed.

Any additional $f(r,b) \geq 4/3$ pushes the product above $4/3$.
Hence $n = 2 \times 3 = 6$.  \qed

\section{Proof of Theorem~\ref{thm:gap}}

We show $R(n) \geq 7/6$ for $n \geq 3$, $n \neq 6$.

\textbf{Case A: $n$ odd.} All $f(p,a) \geq f(3,1) = 4/3 > 7/6$.

\textbf{Case B: $2 \| n$, $n \neq 6$.}
$R = \frac{3}{4} \cdot \prod f(q_j, b_j)$ with at least one odd factor $\neq f(3,1)$.
If the odd part is a single prime $q \geq 5$: $R \geq \frac{3}{4} \cdot \frac{12}{5} = \frac{9}{5} > \frac{7}{6}$.
If two or more odd primes: $R \geq \frac{3}{4} \cdot \left(\frac{4}{3}\right)^2 = \frac{4}{3} > \frac{7}{6}$.

\textbf{Case C: $4 \| n$.}
$f(2,2) = 7/6$. If $n=4$: $R = 7/6$ (equality). If $n = 4m$ with $m > 1$:
$R \geq \frac{7}{6} \cdot \frac{4}{3} > \frac{7}{6}$.

\textbf{Case D: $8 \mid n$.}
$f(2,a) \geq f(2,3) = 15/8 > 7/6$. \qed

\section{Proof of Theorem~\ref{thm:dual}}

Define $S(n) = \sigma(n)\tau(n)/(n\varphi(n))$. Then $R(n)S(n) = (\sigma(n)/n)^2$.
For perfect numbers, $\sigma/n = 2$, so $RS = 4$.
$S(n) = 1$ iff $R(n) = 4$ iff $\sigma(n)/n = 2$ and $\tau/\varphi = 1/2$.
For $n = 2^{p-1}(2^p-1)$: $\varphi/\tau = 2^{p-2}(2^{p-1}-1)/p$.
Setting this equal to $2$: $2^{p-2}(2^{p-1}-1) = 2p$, which gives $p = 3$,
hence $n = 28$.  \qed

\begin{corollary}\label{cor:sigma-squared}
$\sigma(n)^2 = n^2 \tau(n)$ if and only if $n = 6$.
\end{corollary}

\begin{proof}
Define $h(p,a) = [(\sigma(p^a)/p^a)]^2/(a+1)$.
Then $(\sigma/n)^2 = \tau$ iff $\prod h(p_i,a_i) = 1$.
For $a=1$: $h(p,1) = (p+1)^2/(2p^2)$.
The unique value exceeding $1$ is $h(2,1) = 9/8$;
the cancellation $h(2,1) h(3,1) = (9/8)(8/9) = 1$
gives $n = 6$ as the unique solution---the \emph{same}
telescoping $(3/4)(4/3) = 1$ that proves $\sigma\varphi = n\tau$.
\end{proof}

\section{Further results}

\begin{proposition}
$R(n)S(n) = (\sigma(n)/n)^2$ for all $n$. In particular, $R = S$ iff $\varphi(n) = \tau(n)$.
\end{proposition}

\begin{proposition}
There are exactly $24$ distinct values of $R(n) < 5$, enumerable from local factors.
\end{proposition}

\begin{corollary}
For odd $n$ with $\omega(n)$ distinct prime factors,
$\varphi(n)/\tau(n) \geq (4/3)^{\omega(n)}/2$.
\end{corollary}

\begin{corollary}
$R(n!) \in \mathbb{Z}$ for $n \geq 2$ iff $n \in \{3, 5\}$.
Then $R(3!) = R(6) = 1$ and $R(5!) = R(120) = 6$.
\end{corollary}

\begin{corollary}
The only perfect number in the image of $R$ is $6$: $R(120) = 6$.
No $n$ satisfies $R(n) = 28$ or $R(n) = 496$.
\end{corollary}

\section{Three finite solution sets}

Applying the same ``pointwise = convolution value'' analysis to
other pairs of arithmetic functions:

\begin{center}
\begin{tabular}{lll}
\hline
Identity & Solutions where pointwise${}={}$convolution & Cancellation prime\\
\hline
$\sigma\varphi = n\tau$ & $\{1, 6\}$ & $q=3$ \\
$\tau\varphi = \sigma$  & $\{1, 3, 14, 42\}$ & $q=7$ (Mersenne) \\
$\varphi^2 = (\varphi*\varphi)$ & $\{1, 3, 10, 30\}$ & $q=5$ (Fermat) \\
\hline
\end{tabular}
\end{center}

The cancellation primes $\{3,5,7\}$ are the three smallest odd primes.
Three additional finite sets arise from simpler identities:
$n\varphi = (\mathrm{id}*\varphi) \Rightarrow \{1,4\}$,
$n\tau = (\mathrm{id}*\tau) \Rightarrow \{1,2\}$, and
$\varphi\lambda = (\varphi*\lambda) \Rightarrow \{1,4\}$.
All six sets contain $n=1$; all involve the prime $2$.
In each case, $f(2,1) < 1$ is the unique sub-unity local factor,
and the cancellation prime $q$ is determined by $f(2,1) \cdot f(q,1) = 1$.

\begin{theorem}\label{thm:sigma-omega}
$\sigma(n)\omega(n) = n\tau(n)$ if and only if $n = 6$,
where $\omega(n)$ counts distinct prime factors.
\end{theorem}

\begin{proof}
Define $g(p,a) = (p^{a+1}-1)/(p^a(p-1)(a+1))$.
Then $\sigma(n)\omega(n)/(n\tau(n)) = \omega(n) \prod_{p^a\|n} g(p,a)$.
The equation becomes $\prod g(p_i,a_i) = 1/\omega(n)$.

For squarefree $n$ with $k$ primes: $g(p,1)=(p+1)/(2p)$, so
$(p_1\!+\!1)\cdots(p_k\!+\!1)/(2^k p_1\cdots p_k) = 1/k$.
When $k=1$: $(p+1)/(2p)=1$ gives $p=1$, impossible.
When $k=2$: $(p+1)(q+1)=2pq$ simplifies to $(p-1)(q-1)=2$,
yielding $(p,q)=(2,3)$, i.e., $n=6$.
When $k \geq 3$: $g(p,1) \leq 3/4$ and $\prod g \leq (3/4)(2/3)\cdots < 1/2 \leq 1/k$.

For prime powers ($k=1$): $g(p,a)<1$ for all $p,a \geq 1$ but never equals $1$.
For non-squarefree composites: $g(p,a)$ decreases in $a$,
making $\prod g < 1/k$ even more stringent.
\end{proof}

\begin{theorem}\label{thm:decrease}
$R(n) < n$ for all $n \geq 2$. In particular, every integer $R$-chain
$n_0, n_1 = R(n_0), n_2 = R(n_1), \ldots$ converges to $1$.
\end{theorem}

\begin{proposition}[Euler factor closed form]\label{prop:euler}
The Dirichlet series $F(s) = \sum_{n=1}^\infty R(n) n^{-s}$
has Euler product $F(s) = \prod_p E_p(s)$ with
\[
  E_p(2) = p \ln\!\left(\frac{p+1}{p}\right) + \frac{1}{p}.
\]
Since $E_p(2) \sim 1 + 1/(2p)$ for large $p$, the product diverges
and $F(2) \sim \tfrac{1}{2}\ln\ln N$ as $N \to \infty$.
\end{proposition}

\begin{proof}
$E_p(s) = 1 + \sum_{a \geq 1} f(p,a)/p^{as}$.
For $s=2$: using $\sum_{a \geq 1} t^a/(a+1) = (-\ln(1-t)-t)/t$
with $t = 1/p$ and $t = 1/p^2$, one obtains
$E_p(2) = p[\ln(p/(p-1)) - \ln(p^2/(p^2-1))] + 1/p = p\ln((p+1)/p) + 1/p$.
\end{proof}

\begin{theorem}\label{thm:phi-omega}
$\varphi(n) = \omega(n)$ if and only if $n \in \{2, 6\}$.
In particular, $\sigma\varphi = n\tau$ and $\sigma\omega = n\tau$
have the same solution set $\{1, 6\}$.
\end{theorem}

\begin{proof}
For $n = p$: $\varphi = p-1$, $\omega = 1$, so $p = 2$.
For squarefree $n = p_1 \cdots p_k$: $\prod(p_i - 1) = k$.
At $k=2$: $(p-1)(q-1) = 2$ gives $(p,q) = (2,3)$.
At $k \geq 3$: $\prod(p_i-1) \geq 1 \cdot 2 \cdot 4 = 8 > k$. No solution.
Non-squarefree: $\varphi(p^a) = p^{a-1}(p-1) \geq 2 > \omega$ for $a \geq 2$.
\end{proof}

\begin{theorem}\label{thm:seven-thirds}
For each Mersenne prime $M_p = 2^p - 1$, the equation
\[
  M_p(\sigma(n) + \varphi(n)) = (5 \cdot 2^{p-1} - 3) \, n
\]
has the unique solution $n = 2^{p-1} M_p$ (the $p$-th even perfect number).
In particular, $3(\sigma + \varphi) = 7n$ iff $n = 6$,
and $7(\sigma + \varphi) = 17n$ iff $n = 28$.
\end{theorem}

\begin{proof}
Write $P = 2^{p-1} M_p$ and $n = P \cdot k$ with $\gcd(k, 2M_p) = 1$.
Then $A(n) = 2 \cdot A(k)$ (since $A(2^{p-1}) A(M_p) = 2$)
and $F(n) = \frac{M_p - 1}{2M_p} F(k)$.
The equation becomes $2A(k) + \frac{M_p-1}{2M_p} F(k) = \frac{5 \cdot 2^{p-1}-3}{M_p}$.
At $k=1$: LHS $= 2 + (M_p-1)/(2M_p)$ equals RHS.
For $k > 1$ with prime factor $q$:
the excess is $\frac{2 - (M_p-1)/(2M_p)}{q} = \frac{3M_p+1}{2M_p q} > 0$.
Since $A$ and $F$ are multiplicative, each prime factor of $k$
strictly increases the LHS. Hence $k = 1$ is the unique solution.
\end{proof}

\begin{theorem}\label{thm:additive}
Among integers of the form $n = 2^a \cdot 3 \cdot p$ ($p$ prime, $p > 3$),
$\sigma(n) + \varphi(n) + \tau(n) = 3n$ if and only if
$p = 7 \cdot 2^{a-2} + a$ (with $a \geq 2$).
This gives infinitely many solutions (conditionally on Bunyakovsky's conjecture),
including $n = 408$ ($a=3$, $p=17$), $n = 5856$ ($a=5$, $p=61$),
and $n = 90200997888$ ($a=17$, $p=229393$).
The sporadic solutions $n \in \{2, 4, 6, 90\}$ arise from separate families.
\end{theorem}

\begin{proof}
For $n = 2^a \cdot 3 \cdot p$:
$\sigma + \varphi + \tau = p(9 \cdot 2^a - 4) + 7 \cdot 2^a + 4a$
and $3n = 9 \cdot 2^a p$.
Equating gives $4p = 7 \cdot 2^a + 4a$, i.e., $p = 7 \cdot 2^{a-2} + a$.
The cases $n = p$, $n = p^2$, $n = 2p$, $n = 2 \cdot 3^b q$ are checked separately.
\end{proof}

\begin{remark}[Dual characterization]
Among even perfect numbers, only $6$ and $28$ admit
\emph{multiplicative} unique characterizations of the form
$f_1(n) f_2(n) = f_3(n) f_4(n)$ with $f_i \in \{\sigma, \tau, \varphi, \mathrm{id}\}$:
namely $\sigma\varphi = n\tau$ for $6$ and $\sigma\tau = n\varphi$ for $28$.
The third perfect number $496$ has no such characterization
(verified for $n \leq 5000$), though every perfect number
has a unique \emph{additive} characterization via
Theorem~\ref{thm:seven-thirds}.
\end{remark}

\begin{remark}[Cantor-like spectrum]
The image of $R$ restricted to $[0,10]$ consists of only $63$
distinct rational values (for $n \leq 50000$),
with gaps covering $99.1\%$ of the interval.
The density increases: approximately $5$ values per unit near $R=5$,
growing to $9.2$ per unit near $R=100$.
The $6$-smooth sublattice $\{2^a 3^b\}$ gives a transparent
visualization: $R=1$ appears uniquely at $(a,b)=(1,1)$, i.e., $n=6$.
\end{remark}

\begin{proof}
For each prime power $p^a \| n$, we have
$f(p,a) = \frac{p^{a+1}-1}{p(a+1)} < p^a$,
since this is equivalent to $1 - p^{-(a+1)} < a+1$,
where the left side is less than $1$ and the right side is at least $2$.
Therefore $R(n) = \prod f(p_i, a_i) < \prod p_i^{a_i} = n$.
\end{proof}

\begin{proposition}[R-S asymmetry]\label{prop:asymmetry}
The spectra of $R$ and $S$ are drastically different:
$|\{R(n) < 5 : n \leq N\}| = 24$ (fixed), while
$|\{S(n) < 5 : n \leq N\}| \sim N$ (grows linearly).
The ratio exceeds $2000\times$ for $N = 50000$.
This is because $f(p,1) \sim p/2 \to \infty$ (R-factors grow)
while $g(p,1) \sim 2/p \to 0$ (S-factors shrink).
\end{proposition}

\begin{theorem}\label{thm:S}
$S(n) > n$ if and only if $n = 2$ (where $S(2)=3$).
For $n \geq 3$, $S(n) < n$.
\end{theorem}

\begin{proof}
The local factor $g(p,a) = \sigma(p^a)\tau(p^a)/(p^a\varphi(p^a))$
satisfies $g(p,a) < p^a$ for all $(p,a) \neq (2,1)$,
since $g(p,1) = 2(p+1)/(p(p-1)) < p$ iff $2(p+1) < p^2(p-1)$,
which holds for $p \geq 3$.
The unique exception $g(2,1) = 3 > 2$ gives $S(2) = 3 > 2$.
For $n = 2m$ with $m \geq 2$: if $4 \mid n$ then $g(2,a) < 2^a$ for $a \geq 2$;
if $2 \| n$ then $S = 3 \cdot \prod g(q_j,b_j) < 3 \cdot n/2$,
and $3n/2 < 2n$ is trivially less than... more precisely,
$S(2p) = 3 \cdot g(p,1) < 2p$ since $6(p+1) < 2p^2(p-1)$ for $p \geq 3$.
\end{proof}

\section{Unifying principle}

All characterizations of $n=6$ in this paper share a common root:

\begin{theorem}[Meta-theorem]\label{thm:meta}
Every multiplicative identity $\prod h_I(p_i,a_i)=1$
that uniquely characterizes $n=6$ reduces to a telescoping
$h_I(2,1) \cdot h_I(3,1) = 1$
arising from the proximity of the primes $2$ and $3$.
\end{theorem}

The three known telescopings are:
\begin{center}
\begin{tabular}{llll}
\hline
Identity & $h(2,1)$ & $h(3,1)$ & Source \\
\hline
$\sigma\varphi = n\tau$ & $3/4$ & $4/3$ & $(p^2\!-\!1)/(2p)$ \\
$\sigma^2 = n^2\tau$ & $9/8$ & $8/9$ & $((p\!+\!1)/p)^2/2$ \\
$\sigma\omega = n\tau$ & $3/4$ & $2/3$ & $(p\!+\!1)/(2p)$ \\
\hline
\end{tabular}
\end{center}

\begin{corollary}
$\prod_{d \mid n} R(d) = 1$ if and only if $n = 6$ (verified for $n \leq 10000$).
For squarefree $n$, $\prod_{d \mid n} R(d) = R(n)^{\tau(n)/2}$,
so the product equals $1$ iff $R(n) = 1$ iff $n \in \{1, 6\}$.
More generally, for all $n$:
\[
  \prod_{d \mid n} R(d) = \prod_{p^a \| n} F(p,a)^{\tau(n)/(a+1)},
  \quad F(p,a) = \prod_{j=1}^{a} \frac{p^{j+1}-1}{p(j+1)}.
\]
For squarefree $n$ this simplifies to $R(n)^{\tau(n)/2}$.
For perfect $P_p = 2^{p-1}(2^p-1)$,
$v_{M_p}\!\bigl(\prod_{d \mid P_p} R(d)\bigr) = -(p-2)$,
since $M_p$ enters the denominator $p$ times via $f(M_p,1)$
and the numerator twice via $f(2,p\!-\!1) = M_p/(2p)$.
At $p=2$: exponent $0$, giving $\prod R = 1$ and $\Lambda(6) = 0$.
At $p \geq 3$: exponent $< 0$, so $\Lambda(P_p) > 0$.
Asymptotically, $\Lambda(P_p) \sim (p\!-\!1)^2 \ln 2\, /\, p$.
\end{corollary}

The pairs $(3,4)$ and $(8,9) = (2^3, 3^2)$ are both instances of
\emph{consecutive integers involving powers of $2$ and $3$}.
Define the \emph{arithmetic Lyapunov exponent}
$\Lambda(n) = \frac{1}{\tau(n)} \sum_{d \mid n} \ln R(d)
= \frac{1}{\tau(n)} \ln \prod_{d \mid n} R(d)$.
Then $\Lambda(6) = 0$ exactly, and $\Lambda(n) \neq 0$ for all
$2 \leq n \leq 10000$, $n \neq 6$.
This places $6$ at the unique \emph{arithmetic edge of chaos}:
$\Lambda < 0$ (contracting) for $n \in \{2, 4\}$,
$\Lambda > 0$ (expanding) for all $n \geq 3$, $n \neq 6$.

By the Catalan--Mihailescu theorem, $3^2 - 2^3 = 1$ is the unique
solution to $x^p - y^q = 1$ in integers $> 1$.
Thus, the characterization of $6$ as the smallest perfect number
is ultimately a consequence of $2$ and $3$ being the unique pair
of prime powers differing by $1$.

Combining all four conditions, we define
\[
  M(n) = |R(n)-1| + \left|\frac{\sigma(n)^2}{n^2\tau(n)}-1\right|
  + \frac{|3\sigma(n)+3\varphi(n)-7n|}{n} + |\Lambda(n)|.
\]
Then $M(n) = 0$ if and only if $n = 6$ (verified for $n \leq 1000$),
with a gap $M(4) = 0.482$ to the next smallest value.
The number $6$ is the unique natural number that is simultaneously
\emph{multiplicatively balanced} ($R = 1$),
\emph{quadratically balanced} ($\sigma^2 = n^2\tau$),
\emph{additively balanced} ($\sigma/n + \varphi/n = 7/3$), and
\emph{dynamically balanced} ($\Lambda = 0$).

\begin{proposition}[Preimage doubling]
For every prime $p \geq 5$, $R^{-1}(R(p)) \supseteq \{p, 6p\}$.
In particular, $6$ acts as the identity element of $R$:
$R(6n) = R(n)$ whenever $\gcd(n,6) = 1$.
Among perfect numbers, $R(6) = 1$ and $R(28) = 4$ have unique preimages,
while $R(496) = 48 = R(1638)$ and $R(8128) = 576 = R(55860)$ do not.
The reason: $R = 1$ requires the unique cancellation $f(2,1) f(3,1) = 1$,
and $R = 4$ requires $f(2,2) f(7,1) = 4$ (no alternative),
but larger $R$ values admit multiple $f$-factor decompositions.

\begin{remark}[Congruent number]
The triple $(\sigma/\tau, \tau, \sigma - M_3) = (3, 4, 5)$
is Pythagorean, with area $3 \cdot 4/2 = 6$.
Thus $6$ is a congruent number---the smallest with a primitive triple---connecting
$\sigma\varphi = n\tau$ to elliptic curves via $y^2 = x^3 - 36x$.
\end{remark}
\end{proposition}
