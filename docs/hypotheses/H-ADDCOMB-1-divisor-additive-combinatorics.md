from math import gcd
from functools import reduce

n = 6

from collections import Counter

# 2. Digit sums
def digit_sum(n, base):
    s = 0
    while n > 0:
        s += n % base
        n //= base
    return s
for b in [2, 3, 5]:
    assert digit_sum(6, b) == 2 == phi

# 3. Numerical semigroup representations

# 5. Mertens
def mobius(n):
    if n == 1: return 1
    factors = []
    d = 2
    temp = n
    while d*d <= temp:
        if temp % d == 0:
            factors.append(d)
            temp //= d
            if temp % d == 0: return 0
        d += 1
    if temp > 1: factors.append(temp)
    return (-1)**len(factors)

M12 = sum(mobius(k) for k in range(1, 13))
assert M12 == -2 == -phi

# 6. lcm
